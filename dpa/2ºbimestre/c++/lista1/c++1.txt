#include <iostream>
using namespace std;

int main() {

    int numero;
    bool primo = true;

    cout << "Me fale o numero que direi se ele é primo: ";
    cin >> numero;

    if (numero < 2) {
        primo = false;
    } else {

        for (int i = 2; i < numero; i++) {

            if (numero % i == 0) {
                primo = false;
                break;
            }
        } 
    }

    if (primo) {
        cout << "O numero é primo";
    } else {
        cout << "O numero nao é primo";
    }

    return 0;
}

#include <iostream>
using namespace std;

int main() {

    int limite;
    int a = 0, b = 1, proximo;

    cout << "Digite um numero entre 50 e 100: ";
    cin >> limite;

    if(limite < 50 || limite > 100) {
        cout << "Numero invalido!";
    }
    else {

        cout << "Sequencia de Fibonacci:\n";

        while(a <= limite) {
            cout << a << " ";

            proximo = a + b;
            a = b;
            b = proximo;
        }
    }

    return 0;
}

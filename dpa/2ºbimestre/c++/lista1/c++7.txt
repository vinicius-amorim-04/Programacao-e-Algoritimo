#include <iostream>
using namespace std;

int main() {

    int pessoas;
    float altura, soma = 0, media;

    cout << "Quantidade de pessoas: ";
    cin >> pessoas;

    for(int i = 1; i <= pessoas; i++) {

        cout << "Digite a altura da pessoa " << i << ": ";
        cin >> altura;

        soma = soma + altura;
    }

    media = soma / pessoas;

    cout << "Altura media = " << media;

    return 0;
}

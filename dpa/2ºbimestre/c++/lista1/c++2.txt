#include <iostream>
#include <cmath>
using namespace std;

int main() {

    int numero;

    cout << "Digite um numero: ";
    cin >> numero;

    for (int i = 0; i <= numero; i++) {

        int resultado = pow(2, i);

        cout << "2^" << i << " = " << resultado << endl;
    }

    return 0;
}
